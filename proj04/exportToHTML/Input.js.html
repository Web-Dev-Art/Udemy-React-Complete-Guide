1    import React, {useRef, useImperativeHandle } from "react";
2    
3    import classes from "./Input.module.css";
4    
5    const Input = React.forwardRef((props, ref) => {
6         const inputRef = useRef();
7    
8         const activate = () => {
9             inputRef.current.focus();
10        };
11   
12        useImperativeHandle(ref, () => {
13            return {
14              focus: activate
15            };
16        });
17   
18        return (
19           <div className={`${classes.control} ${props.isValid === false ? classes.invalid : ''}`}>
20               <label htmlFor={props.id}>{props.label}</label>
21               <input ref={inputRef} type={props.type} id={props.id} value={props.value} onChange={props.onChange} onBlur={props.onBlur} />
22           </div>
23        )
24   });
25   
26   export default Input;
