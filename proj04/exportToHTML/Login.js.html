Login.js 
1    import React, {useState, useEffect, useReducer, useContext, useRef} from 'react';
2    import Button from "../UI/Button/Button";
3    import Card from "../UI/Card/Card";
4    import AuthContext from "../../context/aut-context";
5    import Input from "../UI/Input/Input";
6    
7    import styles from './Login.module.css'
8    
9    const emailReducer = (state, action) => {
10       if (action.type === 'USER_INPUT') {
11           return { value: action.value, isValid: action.value.includes('@') }
12       }
13   
14       if (action.type === 'INPUT_BLUR') {
15           return { value: state.value, isValid: state.value.includes('@') }
16       }
17   
18       return { value: '', isValid: false }
19   };
20   
21   const passwordReducer = (state, action) => {
22       if (action.type === 'USER_INPUT') {
23           return { value: action.value, isValid: action.value.trim().length > 6};
24       }
25   
26       if (action.type === 'INPUT_BLUR') {
27           return { value: state.value, isValid: state.value.trim().length > 6 }
28       }
29   
30       return { value: '', isValid: false }
31   };
32   
33   const Login = () => {
34       const [formIsValid, setFormIsValid] = useState(false);
35   
36       const [emailState, dispatchEmail] = useReducer(emailReducer, {
37           value: '',
38           isValid: null
39       });
40   
41       const [passwordState, dispatchPass] = useReducer(passwordReducer, {
42          value: '',
43          isValid: null
44       });
45   
46       const authCtx = useContext(AuthContext);
47       const emailInputRef = useRef();
48       const passwordInputRef = useRef();
49   
50       const { isValid: emailIsValid } = emailState;
51       const { isValid: passwordIsValid } = passwordState;
52   
53       useEffect(() => {
54           const identifier = setTimeout(() => {
55               setFormIsValid(emailIsValid && passwordIsValid);
56           }, 500);
57   
58           return () => {
59               clearTimeout(identifier);
60           }
61       }, [emailIsValid, passwordIsValid]);
62   
63       const emailChangeHandler = (event) => {
64           dispatchEmail({type: 'USER_INPUT', value: event.target.value});
65       };
66   
67       const passwordChangeHandler = (event) => {
68           dispatchPass({type: 'USER_INPUT', value: event.target.value});
69       };
70   
71       const validateEmailHandler = () => {
72           dispatchEmail({type: 'INPUT_BLUR'});
73       };
74   
75       const validatePasswordHandler = () => {
76           dispatchPass({type: 'INPUT_BLUR'});
77       };
78   
79       const submitHandler = (event) => {
80           event.preventDefault();
81           if (formIsValid) {
82               authCtx.onLogin(emailState.value, passwordState.value);
83           }
84           else if (!emailIsValid) {
85               emailInputRef.current.focus();
86           }
87           else {
88               passwordInputRef.current.focus();
89           }
90       };
91   
92       return (
93           <Card className={styles.login}>
94               <form onSubmit={submitHandler}>
95                   <Input type='email' ref={emailInputRef} id='email' label='E-Mail' isValid={emailIsValid} value={emailState.value} onChange={emailChangeHandler} onBlur={validateEmailHandler} />
96                   <Input type='password' ref={passwordInputRef} id='password' label='Password' isValid={passwordIsValid} value={passwordState.value}  onChange={passwordChangeHandler} onBlur={validatePasswordHandler} />
97                   <div className={styles.actions}>
98                       <Button type="submit">Login</Button>
99                   </div>
100              </form>
101          </Card>
102      );
103  };
104  
105  export default Login;
