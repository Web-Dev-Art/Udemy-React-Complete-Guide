MealItemForm.js 
1    import React, {useState, useRef} from 'react';
2    import Input from "../../UI/Input";
3    
4    import classes from './MealItemForm.module.css';
5    
6    
7    const MealItemForm = props => {
8        const [amountIsValid, setAmountIsValid] = useState(true);
9        const amountInputRef = useRef();
10   
11       const submitHandler = (event) => {
12           event.preventDefault();
13   
14           const enteredAmount = +amountInputRef.current.value;
15   
16           if (enteredAmount < 1 || enteredAmount > 5) {
17               setAmountIsValid(false);
18               return;
19           }
20   
21           props.onAddToCart(enteredAmount);
22       };
23   
24       return (
25           <form className={classes.form} onSubmit={submitHandler}>
26               <Input label="Amount" input={{type: 'number', id: 'amount', min: '1', max: '5', step: '1', defaultValue: '1'}} ref={amountInputRef} />
27               <button>+ Add</button>
28               {!amountIsValid && <p>Please enter a valid amount (1-5).</p>}
29           </form>
30       );
31   };
32   
33   export default MealItemForm;
