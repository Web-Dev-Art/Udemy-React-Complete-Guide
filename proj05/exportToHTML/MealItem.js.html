MealItem.js 
1    import React, {useContext} from 'react';
2    import MealItemForm from "./MealItemForm";
3    import CartContext from "../../../store/cart-context";
4    
5    import classes from './MealItem.module.css';
6    
7    const MealItem = (props) => {
8        const cartCtx = useContext(CartContext);
9        const price = `$${props.price.toFixed(2)}`;
10   
11       const addToCartHandler = amount => {
12           cartCtx.addItem({
13               id: props.id,
14               name: props.name,
15               amount: amount,
16               price: props.price
17           })
18       };
19   
20       return (
21           <li className={classes.meal}>
22               <div>
23                   <h3>{props.title}</h3>
24                   <div className={classes.description}>{props.description}</div>
25                   <div className={classes.price}>{price}</div>
26               </div>
27               <div>
28                   <MealItemForm onAddToCart={addToCartHandler}/>
29               </div>
30           </li>
31       );
32   };
33   
34   export default MealItem;
